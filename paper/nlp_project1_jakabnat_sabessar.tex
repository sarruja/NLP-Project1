% ─── Abstract ────────────────────────────────
\begin{abstract}
\joint{
We present an experimental pipeline for binary hate speech classification
using the Kaggle Jigsaw Toxic Comment dataset \citep{jigsaw2018}.
Our pipeline systematically evaluates the impact of different preprocessing
strategies, feature extraction methods, and classifiers on model performance.
We run eight experiments while varying one component at a time, measuring
F1 Macro as the primary metric to account for the highly imbalanced class
distribution (89.8\% non-hate-speech).
Our results show that increasing the TF-IDF vocabulary size from 1{,}000 to
5{,}000 features yields the best performance (F1 Macro = 0.8655), while SVM
consistently outperforms Logistic Regression and Naive Bayes.
}
\end{abstract}

% ─── 1. Introduction ─────────────────────────
\section{Introduction}

\authorone{
Automated hate speech detection is a critical NLP task in content moderation.
Building a reliable classifier requires careful choices at each pipeline stage:
preprocessing, feature extraction, and model selection.
In this paper, we present a modular experimental pipeline that systematically
evaluates these choices in isolation on the Kaggle Jigsaw Toxic Comment dataset
\citep{jigsaw2018}, with the goal of understanding which components most affect
classification performance.
}

% ─── 2. Data ─────────────────────────────────
\section{Data}

\joint{
The dataset consists of 159{,}571 Wikipedia comments, each labelled with one or more of the following categories: \textit{toxic, severe\_toxic, obscene, threat, insult, identity\_hate}, or none. 
We binarize the labels: any comment with at least one hate-related label is classified as \textit{hate speech} (1), otherwise as \textit{not hate speech} (0).\newline
The dataset is highly imbalanced: 89.8\% of comments are non-hate-speech (143{,}346 samples) and only 10.2\% are hate speech (16{,}225 samples), as shown in Appendix Figure~\ref{fig:labels}.
This imbalance motivates our use of F1 Macro as the primary evaluation metric, which equally weights performance on both classes regardless of their frequency.
We also apply class-weighted training (\texttt{class\_weight=`balanced'}) to compensate for the skew.\newline
Prior to all experiments, the duplicate comments were removed, reducing the dataset by a small margin. Specifically, 1,225 duplicate comments were removed, resulting in 158,346 unique samples.
The data was split into 80\% training and 20\% test sets with stratification to preserve the class distribution. 
}\authorone{
We performed outlier detection on text lengths using the IQR method (Appendix, Figure~\ref{fig:outlier}). The upper boundary was approximately 138 words; around 8\% of comments exceed this threshold. Outliers were retained since many represent legitimate long-form comments rather than noise.
}

% ─── 3. Methods ──────────────────────────────
\section{Methods}

\subsection{Pipeline Overview}

\authorone{
Our pipeline consists of three interchangeable stages: (1) preprocessing,
(2) feature extraction, and (3) model training.
Each experiment varies exactly one stage while keeping the other two fixed
at their baseline values, following the one-variable-at-a-time principle.
The baseline configuration (E1) uses stemming + stopword removal, TF-IDF
with 1{,}000 features, and a Support Vector Machine (SVM).
}

\subsection{Preprocessing}

\authortwo{
We implement four configurations.
\textbf{Stem + Stopwords} (E1, baseline): tokenization, stopword removal,
and Snowball stemming.
\textbf{No preprocessing} (E2): raw lowercased tokens only.
\textbf{Stopwords only} (E3): stopword removal without stemming.
\textbf{Stemming only} (E4): stemming without stopword removal.
Numbers are removed in all configurations.
}

\newpage
\subsection{Feature Extraction}

\authorone{
We compare two methods.
\textbf{TF-IDF} \citep{manning2008} weights terms by inverse document
frequency (E1--E4, E6--E8).
\textbf{Bag-of-Words (BoW)} uses raw term counts (E5).
We evaluate vocabulary sizes of 1{,}000 (E1--E5, E7, E8) and 5{,}000 (E6).
}

\subsection{Models}

\authortwo{
We evaluate three classifiers.
\textbf{SVM} \citep{joachims1998} with RBF kernel and class weighting (E1).
\textbf{Logistic Regression} with L2 regularization (E7).
\textbf{Naive Bayes} Multinomial (E8).
We additionally vary the SVM regularization parameter
$C \in \{0.1, 1, 10\}$.
}

\subsection{Evaluation Metric}

\joint{
We report \textbf{F1 Macro}, the unweighted average of per-class F1 scores,
which equally penalizes poor performance on either class.
We additionally report \textbf{F1 Hate Speech} (F1 for the positive class only).
}

% ─── 5. Discussion ───────────────────────────
\section{Discussion}

\subsection{Effect of Preprocessing}

\authortwo{
Comparing E1--E4 reveals that stemming contributes more to performance than
stopword removal: removing only stopwords (E3: 0.7979) performs worse than
stemming only (E4: 0.8238).
Interestingly, no preprocessing (E2: 0.8244) is close to the baseline
(E1: 0.8316), indicating that SVM is relatively robust to raw text.
}

\subsection{Effect of Feature Extraction}

\authorone{
Increasing the TF-IDF vocabulary from 1{,}000 to 5{,}000 features (E6)
improves F1 Macro by 3.4 percentage points, suggesting that hate speech
detection benefits from a richer vocabulary.
BoW (E5: 0.7848) underperforms TF-IDF (E1: 0.8316), confirming that
inverse document frequency weighting is beneficial \citep{manning2008}.
}

\subsection{Effect of Model Choice}

\authortwo{
SVM (E1: 0.8316) outperforms both Logistic Regression (E7: 0.7725) and
Naive Bayes (E8: 0.7907).
The SVM's maximum-margin boundary is particularly effective in
high-dimensional TF-IDF spaces \citep{joachims1998}.
}

\subsection{Hyperparameter Sensitivity}

\authorone{
The SVM regularization experiment shows that $C=1$ achieves the best F1 Macro (0.8316), while $C=0.1$
(0.8189) and $C=10$ (0.8222) perform slightly worse, indicating robustness
to this parameter in the given range.
}

% ─── 6. Conclusion ───────────────────────────

\section{Conclusion}

\joint{
We developed a modular experimental pipeline for binary hate speech
classification and systematically evaluated eight configurations.
Our main findings are: (1) increasing TF-IDF vocabulary to 5{,}000 features
yields the largest gain (+3.4 pp F1 Macro); (2) SVM outperforms Logistic
Regression and Naive Bayes; (3) stemming is more beneficial than stopword
removal; and (4) SVM is robust to the regularization parameter $C$.
Future work could explore larger vocabularies or pre-trained language models
such as BERT.
}
